# ex26: World Population Trends
```
#
# data from:
#  https://www.worldometers.info/world-population/world-population-by-year/
#

read year,n,dndy,dn,dnds `sed "s/,//g" world-population-20260105.txt`

opt (gd:1 lc:red)

vp 0 0 0 0 0.7
div 2 2
ylab "Population"
plot year n (ys:log yr:1e8,1e10)

ylab "Yearly Change (%)"
plot year dndy (xr:1951,2025)

ylab "Net Change"
plot year dn (xr:1951,2025 ys:log yr:1e7,3e8)

ylab "Population / km#u2#d"
plot year dnds (ys:lin)

vp 0 1 0 1 0
div 3 18
box 0 10 0 10 (lw:0 bp:2)
text 1 5 "World Population Trends" (ts:4)
```
